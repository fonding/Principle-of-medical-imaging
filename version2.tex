\begin{equation}
S(x,z,\alpha) = \displaystyle{\int RF[x_{1},t(x,z,\alpha)] dx_{1} }
\end{equation}
\begin{equation}
CI(x,z) = \sum_{i=1}^nS(x,z,\alpha_{i})
\end{equation}or
\begin{equation}
CI = \frac{1}{N}\sum_{i=1}^NS_{B}^{2}(t_{i})
\end{equation}
where $\alpha$ is the angle of deflection.\\ $CI$ represent final combined images.
\end{frame}
